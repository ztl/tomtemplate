  <div class="md" id="tabview">
  	<div class="md-bd">
  		<div class="ui-tab">
  			<ul>
  				<li class="active"><a href="#">新闻</a></li>
  				<li><a href="#">图片</a></li>
  				<li><a href="#">军事</a></li>
  			</ul>
  		</div>
  		<div class="view">
  			省略区块1
  		</div>
  		<div class="view hide">
  			省略区块2
  		</div>
  		<div class="view hide">
  			省略区块3
  		</div>
  	</div>
  </div>

  <script src="./static/js/common/require.js"></script>
  <script>
  require.config({
  	baseUrl: '.././static/js',
  	paths: {
  		'jquery'	: 'common/jquery-1.8.3.min',
  		'Tab'		: 'common/tab'
  	}
  });

  require(['jquery', 'Tab'], function($, Tab) {

  	// Tab切换
  	Tab('#tabview .ui-tab li', '#tabview .view', {
  		'trigger': 'mouseenter'
  	});

  });
  </script>

    弹层 Popup

    弹层。

    Demo：

    弹层 1 弹层 2

    代码：

      <link rel="stylesheet" href="./static/css/ui/popup.css" />

  <script type="text/tpl" id="layer-1" style="display:none;">
  	<h2>标题</h2>
  	<p>Layer 1</p>
  	<p><a href="#" id="popup-3">弹层 3</a></p>
  </script>
  <script type="text/tpl" id="layer-2" style="display:none;">
  	<p>Layer 2</p>
  </script>
  <script type="text/tpl" id="layer-3" style="display:none;">
  	<p>Layer 3</p>
  	<p><img src="../static/img/product/pic10.jpg" /></p>
  </script>

  <script src="./static/js/common/popup.js"></script>
  <script>
  // 实例化弹层
  var popup = new Popup();

  // 弹层 1
  $('#popup-1').on('click', function(evt) {
  	var renderHtml = $('#layer-1').html();
  	popup.renderBody(renderHtml).show();
  	evt.preventDefault();
  });

  // 弹层 2
  $('#popup-2').on('click', function(evt) {
  	var renderHtml = $('#layer-2').html();
  	popup.renderBody(renderHtml).show();
  	evt.preventDefault();
  });

  // 弹层 3
  $('.ui-popup').delegate('#popup-3', 'click', function(evt) {
  	var renderHtml = $('#layer-3').html();
  	popup.renderBody(renderHtml).show();
  	evt.preventDefault();
  });
  </script>
